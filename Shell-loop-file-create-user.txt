break : to stop or terminate the iterator 
continue : skip the current iteration and continue from next 

Exmaple: break 

i=1  #init

while [ '$i' -le 10 ] #condition 
do
	mod=`expr $i % 5`
	if [ '$mod' -eq 0 ]
	then
		break 
	fi
	echo "$i"	

	i=`expr $i + 1`  #increment
done
